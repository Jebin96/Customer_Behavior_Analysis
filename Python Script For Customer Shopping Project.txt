import pandas as pd

df = pd.read_csv('customer_shopping_behavior.csv')

df.head()

df.info()

df.describe()

df.describe(include='all')  #---Numerical and Categorical it will give the statistics of describe

df.isnull().sum()  # Finding the null values

df['Review Rating'] = df.groupby('Category')['Review Rating'].transform(lambda x: x.fillna(x.median()))

df.isnull().sum()

df.columns = df.columns.str.lower() # Making the lower in all column name
df.columns = df.columns.str.replace(' ', '_') # replacing space with underscore
df = df.rename(columns = {'purchase_amount_(usd)':'purchase_amount'})

df.columns

# Feature Engineering

#Create a column age_group
labels = ['Young Adult', 'Adult', 'Middle_aged','Senior']
df['age_group'] = pd.qcut(df['age'], q=4, labels = labels) # qcut it will spilts the age group

df[['age','age_group']].head(15)

# Create Column Purchase Frequency Days

Frequency_mapping = {
    'Fortnightly' : 14,
    'Weekly' : 7,
    'Monthly' : 30,
    'Quaterly' : 90,
    'Annually' : 365,
    'Every 3 months' : 90
}

df['purchase_frequency_days'] = df['frequency_of_purchases'].map(Frequency_mapping)
# Mapping the frequency mapping with frequency_of_purchases

df[['purchase_frequency_days','frequency_of_purchases']].head()
